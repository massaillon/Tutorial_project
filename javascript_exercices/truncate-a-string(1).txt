//The goal of the function is to see if the length of the given string is greater than the given maximum string length (num). If it is, truncate the string right at the maximum length and return it with an ellipsis (â€¦) at the end. If the string is shorter or equal to the cutoff string length, return the string as is.

function truncateString(str, num) {
  if (str.length > num)
  {
    let subStr = str.substring(0,num);
    return subStr + "...";
  }
  else
  {
    return str;
  }
  
}

console.log(truncateString("A-tisket a-tasket A green and yellow basket", 18));
